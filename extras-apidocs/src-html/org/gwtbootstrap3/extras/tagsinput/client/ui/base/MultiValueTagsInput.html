023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.Collection;
026import java.util.Collections;
027import java.util.List;
028
029import org.gwtbootstrap3.extras.typeahead.client.base.CollectionDataset;
030import org.gwtbootstrap3.extras.typeahead.client.base.Dataset;
031
032import com.google.gwt.dom.client.Document;
033import com.google.gwt.dom.client.Element;
034import com.google.gwt.dom.client.SelectElement;
035import com.google.gwt.event.logical.shared.HasValueChangeHandlers;
036import com.google.gwt.event.logical.shared.ValueChangeEvent;
037import com.google.gwt.event.logical.shared.ValueChangeHandler;
038import com.google.gwt.event.shared.HandlerRegistration;
039
040/**
041 * Wrapper for Bootstrap Tags Input component implemented with &lt;select&gt; as underlying component.
042 *
043 * @author Marko NikoliÄ‡ <marko.nikolic@iten.rs>
044 */
045public class MultiValueTagsInput<T> extends TagsInputBase<T> implements HasValueChangeHandlers<List<String>> {
046
047    public MultiValueTagsInput() {
048        this(new CollectionDataset<T>(Collections.<T>emptyList()));
049    }
050
051    public MultiValueTagsInput(final Dataset<T> dataset) {
052        this(Arrays.asList(dataset));
053        
054        setDatasets(dataset);
055    }
056
057    public MultiValueTagsInput(final Collection<? extends Dataset<T>> datasets) {
058        SelectElement tagsSelect = Document.get().createSelectElement();
059        tagsSelect.setMultiple(true);
060        tagsSelect.setAttribute("data-role", "tagsinput");
061
062        setElement(tagsSelect);
063        
064        setDatasets(datasets);
065    }
066
067    public MultiValueTagsInput(Element e) {
068        this(e, new CollectionDataset<T>(Collections.<T>emptyList()));
069    }
070
071    public MultiValueTagsInput(Element e, final Dataset<T> dataset) {
072        this(e, Arrays.asList(dataset));
073        
074        setDatasets(dataset);
075    }
076
077    public MultiValueTagsInput(Element e, final Collection<? extends Dataset<T>> datasets) {
078        e.setAttribute("data-role", "tagsinput");
079
080        setElement(e);
081        
082        // Wrapped elements are already attached to the DOM and the onAttach method will not be called automatically,
083        // so it is called manually to correctly set attached state.
084        onAttach();
085        
086        setDatasets(datasets);
087    }
088    
089    /**
090     * Returns comma delimited string with values from tags.
091     * 
092     * @return comma delimited string
093     */
094    public List<String> getValue() {
095        if (isAttached())
096            return toMultiValue(getValue(getElement()));
097        else    
098            return new ArrayList<String>();
099    }
100
101    @Override
102    public HandlerRegistration addValueChangeHandler(ValueChangeHandler<List<String>> handler) {
103        return addHandler(handler, ValueChangeEvent.getType());
104    }
105}
