023import com.google.gwt.user.client.ui.HasText;
024import org.gwtbootstrap3.client.ui.base.AbstractInputGroupAddon;
025import org.gwtbootstrap3.client.ui.base.mixin.IconTextMixin;
026import org.gwtbootstrap3.client.ui.constants.IconPosition;
027import org.gwtbootstrap3.client.ui.constants.IconSize;
028import org.gwtbootstrap3.client.ui.constants.IconType;
029import org.gwtbootstrap3.client.ui.constants.Styles;
030
031/**
032 * Prepends or appends texts to input fields.
033 * <p/>
034 * For prepending/appending {@link Button} see {@link InputGroupButton}.
035 * Needs to be encapsulated in an {@link InputGroup}.
036 * <h3>UiBinder example</h3>
037 * <pre>{@code
038 *     <b:InputGroup>
039 *         <b:InputGroupAddon>Prepend something</b:InputGroupAddon>
040 *         <b:TextBox/>
041 *         <b:InputGroupAddon>Append something</b:InputGroupAddon>
042 *     </b:InputGroup>
043 * }</pre>
044 * <p/>
045 * Also see Bootstrap <a href="http://getbootstrap.com/components/#input-groups">documentation</a>.
046 *
047 * @author Sven Jacobs
048 * @see InputGroup
049 * @see InputGroupButton
050 */
051public class InputGroupAddon extends AbstractInputGroupAddon implements HasText, HasIcon, HasIconPosition, HasIconSize {
052
053    IconTextMixin<InputGroupAddon> iconTextMixin = new IconTextMixin<InputGroupAddon>(this);
054
055    public InputGroupAddon() {
056        super(Styles.INPUT_GROUP_ADDON);
057
058        iconTextMixin.addTextWidgetToParent();
059    }
060
061    @Override
062    public void setText(final String text) {
063        iconTextMixin.setText(text);
064    }
065
066    @Override
067    public String getText() {
068        return iconTextMixin.getText();
069    }
070
071    @Override
072    public void setIcon(final IconType iconType) {
073        iconTextMixin.setIcon(iconType);
074    }
075
076    @Override
077    public IconType getIcon() {
078        return iconTextMixin.getIcon();
079    }
080
081    @Override
082    public void setIconPosition(final IconPosition iconPosition) {
083        iconTextMixin.setIconPosition(iconPosition);
084    }
085
086    @Override
087    public IconPosition getIconPosition() {
088        return iconTextMixin.getIconPosition();
089    }
090
091    @Override
092    public void setIconSize(final IconSize iconSize) {
093        iconTextMixin.setIconSize(iconSize);
094    }
095
096    @Override
097    public IconSize getIconSize() {
098        return iconTextMixin.getIconSize();
099    }
100}
