023import com.google.gwt.dom.client.Document;
024import com.google.gwt.dom.client.InputElement;
025import com.google.gwt.event.dom.client.*;
026import com.google.gwt.event.shared.HandlerRegistration;
027import com.google.gwt.uibinder.client.UiConstructor;
028import com.google.gwt.user.client.ui.HasEnabled;
029import com.google.gwt.user.client.ui.HasName;
030import org.gwtbootstrap3.client.ui.base.ComplexWidget;
031import org.gwtbootstrap3.client.ui.constants.ElementTags;
032import org.gwtbootstrap3.client.ui.constants.InputType;
033import org.gwtbootstrap3.client.ui.constants.Styles;
034
035/**
036 * @author Joshua Godi
037 */
038public class Input extends ComplexWidget implements HasEnabled, HasInputType, HasPlaceholder, HasFormValue, HasName, HasAllKeyHandlers {
039    private static final String MIN = "min";
040    private static final String MAX = "max";
041    private static final String DISABLED = "disabled";
042
043    public Input() {
044        setElement(Document.get().createElement(ElementTags.INPUT));
045        addStyleName(Styles.FORM_CONTROL);
046    }
047
048    @UiConstructor
049    public Input(final InputType type) {
050        this();
051        setType(type);
052    }
053
054    public void setMin(final String min) {
055        getElement().setAttribute(MIN, min);
056    }
057
058    public void setMax(final String max) {
059        getElement().setAttribute(MAX, max);
060    }
061
062    @Override
063    public void setEnabled(final boolean enabled) {
064        getElement().setPropertyBoolean(DISABLED, !enabled);
065    }
066
067    @Override
068    public boolean isEnabled() {
069        return !getElement().getPropertyBoolean(DISABLED);
070    }
071
072    @Override
073    public void setType(final InputType inputType) {
074        getElement().setAttribute(TYPE, inputType.getType());
075    }
076
077    @Override
078    public InputType getType() {
079        if (getElement().getAttribute(TYPE) == null || getElement().getAttribute(TYPE).isEmpty()) {
080            return null;
081        }
082        return InputType.valueOf(getElement().getAttribute(TYPE));
083    }
084
085    @Override
086    public void setPlaceholder(final String placeHolder) {
087        getElement().setAttribute(PLACEHOLDER, placeHolder != null ? placeHolder : "");
088    }
089
090    @Override
091    public String getPlaceholder() {
092        return getElement().getAttribute(PLACEHOLDER);
093    }
094
095    @Override
096    public String getFormValue() {
097        return InputElement.as(getElement()).getValue();
098    }
099
100    @Override
101    public void setFormValue(final String value) {
102        InputElement.as(getElement()).setValue(value);
103    }
104
105    @Override
106    public String getName() {
107        return InputElement.as(getElement()).getName();
108    }
109
110    @Override
111    public void setName(final String name) {
112        InputElement.as(getElement()).setName(name);
113    }
114
115    @Override
116    public HandlerRegistration addKeyDownHandler(KeyDownHandler handler) {
117        return addDomHandler(handler, KeyDownEvent.getType());
118    }
119
120    @Override
121    public HandlerRegistration addKeyPressHandler(KeyPressHandler handler) {
122        return addDomHandler(handler, KeyPressEvent.getType());
123    }
124
125    @Override
126    public HandlerRegistration addKeyUpHandler(KeyUpHandler handler) {
127        return addDomHandler(handler, KeyUpEvent.getType());
128    }
129}
