023import java.util.Arrays;
024import java.util.Collection;
025import java.util.Collections;
026
027import org.gwtbootstrap3.extras.typeahead.client.base.CollectionDataset;
028import org.gwtbootstrap3.extras.typeahead.client.base.Dataset;
029
030import com.google.gwt.dom.client.Document;
031import com.google.gwt.dom.client.Element;
032import com.google.gwt.dom.client.InputElement;
033import com.google.gwt.event.logical.shared.HasValueChangeHandlers;
034import com.google.gwt.event.logical.shared.ValueChangeEvent;
035import com.google.gwt.event.logical.shared.ValueChangeHandler;
036import com.google.gwt.event.shared.HandlerRegistration;
037
038/**
039 * Wrapper for Bootstrap Tags Input component implemented with &lt;input&gt; box as underlying component.
040 *
041 * @author Marko NikoliÄ‡ <marko.nikolic@iten.rs>
042 */
043public class SingleValueTagsInput<T> extends TagsInputBase<T> implements HasValueChangeHandlers<String> {
044
045    public SingleValueTagsInput() {
046        this(new CollectionDataset<T>(Collections.<T>emptyList()));
047    }
048
049    public SingleValueTagsInput(final Dataset<T> dataset) {
050        this(Arrays.asList(dataset));
051        
052        setDatasets(dataset);
053    }
054
055    public SingleValueTagsInput(final Collection<? extends Dataset<T>> datasets) {
056        InputElement tagsInput = Document.get().createTextInputElement();
057        tagsInput.setAttribute("data-role", "tagsinput");
058        
059        setElement(tagsInput);        
060        
061        setDatasets(datasets);
062    }
063 
064    public SingleValueTagsInput(final Element e) {
065        this(e, new CollectionDataset<T>(Collections.<T>emptyList()));
066    }
067
068    public SingleValueTagsInput(final Element e, final Dataset<T> dataset) {
069        this(e, Arrays.asList(dataset));
070        
071        setDatasets(dataset);
072    }
073
074    public SingleValueTagsInput(final Element e, final Collection<? extends Dataset<T>> datasets) {
075        e.setAttribute("data-role", "tagsinput");
076
077        setElement(e);
078        
079        // Wrapped elements are already attached to the DOM and the onAttach method will not be called automatically,
080        // so it is called manually to correctly set attached state.
081        onAttach();
082        
083        setDatasets(datasets);
084    }
085    
086    /**
087     * Returns comma delimited string with values from tags.
088     * 
089     * @return comma delimited string
090     */
091    public String getValue() {
092        if (isAttached())
093            return getValue(getElement()).toString();
094        else
095            return null;
096    }
097
098    @Override
099    public HandlerRegistration addValueChangeHandler(ValueChangeHandler<String> handler) {
100        return addHandler(handler, ValueChangeEvent.getType());
101    }
102}
